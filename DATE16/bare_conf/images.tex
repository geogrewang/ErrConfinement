\begin{algorithm}
% latex2html id marker 352
[hbt]                    % enter the algorithm environment
\caption{Token Tracking Routine}          % give the algorithm a caption
{\fontsize{8}{8}\selectfont
\begin{algorithmic}[1]
\Function{trackToken}{$*op,*token,*PeMM$}
    \ForAll{ $op\_id$\  }  \Comment{Create tokens by activation analysis}
       \If {$op[op\_id]\ is\ active$}
          \If {$\exists token[tk\_id]$\  in $op[op\_id].inputs$}
          \State Update with $PeMM[op\_id]$\  for $op[op\_id]$;
          \State Schedule to create tokens in $op[op\_id].outputs$;
          \State New tokens labeled as $high$\  priority;
          \EndIf
       \EndIf
    \EndFor
    \ForAll{ $tk\_id$\  } \Comment{Create tokens by pipeline behaviors}
              \If {$token[tk\_id]$\  is in pipeline registers}
              \State Schedule to remove $token[tk\_id]$;
                  \If {$token[tk\_id]$\  is \textbf{not} in last pipeline stage}
                  \State Schedule forwarding token in next stage as $low$\  priority;
                  \EndIf
              \EndIf
    \EndFor
    \State Remove\_tokens();
    \State Create\_tokens(); \Comment{Create/remove tokens at end of the control step}
\EndFunction
   \vspace{0.2cm}
\Function{Create\_tokens}{}
    \ForAll{ $tokens$\  in $schedule\ creating\ list$}
       \If {$\exists old\ token$\  in $new\ location$}
              \State Overwrite $old\ token$;
       \EndIf  \Comment{Overwrite existing tokens}
       \If {multiple tokens are scheduled in the same location}
              \State Create token with $high$\  priority;
       \Else   \Comment{Forwarded tokens have less priority}
              \State Create token;
       \EndIf
    \EndFor
\EndFunction
\end{algorithmic}
}
\end{algorithm}%
